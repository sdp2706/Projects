        "# creating a the health care bot\n",
        "def chat_with_hena(selected_symptoms, history=[]):\n",
        "    if not selected_symptoms:\n",
        "        return \"No symptoms selected. Please choose at least one symptom.\", history, None\n",
        "\n",
        "    combined_advice = []\n",
        "    for symptom in selected_symptoms:\n",
        "        if symptom in symptom_data:\n",
        "            data = symptom_data[symptom]\n",
        "            combined_advice.append(f\"For {symptom}, {data['advice']} You can use {data['medicines']}.\")\n",
        "\n",
        "    combined_response = \"Hey there! \" + \" \".join(combined_advice) + \" If you need more help, just let me know!\"\n",
        "\n",
        "    history.append((', '.join(selected_symptoms), combined_response))\n",
        "\n",
        "    # Convert the response to speech\n",
        "    tts = gTTS(combined_response, lang='en')\n",
        "    audio_file = \"chatbot_response.mp3\"\n",
        "    tts.save(audio_file)\n",
        "\n",
        "    return combined_response, history, audio_file"
